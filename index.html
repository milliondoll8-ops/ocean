"ПОВЫСИТЬ УРОВЕНЬ" : "ВСТУПИТЬ В СЕМЬЮ (10 USDT)"; const total = await con.lastUserId(); document.getElementById('totalUsers').innerText = total.toString(); } catch(e) { console.error("Ошибка загрузки данных контракта", e); } } async function handleAction() { try { const signer = provider.getSigner(); const usdt = new ethers.Contract(USDT_ADDR, ["function approve(address,uint256) public returns(bool)"], signer); const amount = ethers.utils.parseUnits("10.0", 6); document.getElementById('footerStatus').innerText = "1/2 ПОДТВЕРДИТЕ USDT..."; const appTx = await usdt.approve(CONTRACT, amount); await appTx.wait(); document.getElementById('footerStatus').innerText = "2/2 РЕГИСТРАЦИЯ..."; const myCon = new ethers.Contract(CONTRACT, MY_ABI, signer); const urlParams = new URLSearchParams(window.location.search); const ref = urlParams.get('ref') || OWNER; const tx = await myCon.registration(ref); await tx.wait(); alert("УСПЕХ!"); location.reload(); } catch (e) { console.error(e); alert("Ошибка транзакции! Убедитесь, что у вас есть MATIC для газа и 10 USDT."); document.getElementById('footerStatus').innerText = ""; } } function copyLink() { const i = document.getElementById("refLink"); i.select(); navigator.clipboard.writeText(i.value); alert("Ссылка скопирована!"); }
