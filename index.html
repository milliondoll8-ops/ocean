// --- КОНФИГУРАЦИЯ --- const CONTRACT_ADDR = "ТВОЙ_АДРЕС_КОНТРАКТА"; const USDT_ADDR = "0xc2132D05D31c914a87C6611C10748AEb04B58e8F"; const ABI = [ "function register(address _referrer) external", "function buyLevel(uint8 _level) external", "function users(address) view returns (uint256 id, address referrer, uint8 activeLevel, uint256 personalReferralsCount)", "function owner() view returns (address)", "function withdrawLeaderPool() external", "function leaderPool() view returns (uint256)" ]; const USDT_ABI = [ "function approve(address spender, uint256 amount) public returns (bool)", "function balanceOf(address) view returns (uint256)", "function allowance(address owner, address spender) view returns (uint256)" ]; let provider, signer, contract, usdtContract; // ГЛАВНАЯ ФУНКЦИЯ КОННЕКТА async function connect() { console.log("Попытка подключения..."); if (typeof window.ethereum !== 'undefined') { try { // Прямой запрос разрешений у Метамаска const accounts = await window.ethereum.request({ method: 'eth_requestAccounts' }); console.log("Аккаунты получены:", accounts); provider = new ethers.BrowserProvider(window.ethereum); signer = await provider.getSigner(); const userAddress = await signer.getAddress(); // Инициализация контрактов contract = new ethers.Contract(CONTRACT_ADDR, ABI, signer); usdtContract = new ethers.Contract(USDT_ADDR, USDT_ABI, signer); // Визуальное обновление интерфейса document.getElementById('connectBtn').innerText = "Подключен: " + userAddress.slice(0,6) + "..."; document.getElementById('userAddr').innerText = userAddress; // Загрузка данных await loadUserData(userAddress); } catch (error) { console.error("Ошибка при подключении:", error); if (error.code === 4001) { alert("Вы отклонили запрос на подключение кошелька."); } else { alert("Ошибка: " + error.message); } } } else { alert("Метамаск не найден! Установите расширение или используйте крипто-браузер."); } } async function loadUserData(addr) { try { const userData = await contract.users(addr); const ownerAddr = await contract.owner(); const poolAmount = await contract.leaderPool(); // Если это ТЫ (Owner/ID 1) if (addr.toLowerCase() === ownerAddr.toLowerCase()) { document.getElementById('u-id').innerText = "1 (FOUNDER)"; document.getElementById('u-level').innerText = "15 (MAX)"; document.getElementById('adminPanel').classList.remove('hidden'); // Показываем сколько в пуле специально для тебя console.log("В лидерском пуле сейчас:", ethers.formatUnits(poolAmount, 6), "USDT"); } else { document.getElementById('u-id').innerText = userData.id.toString(); document.getElementById('u-level').innerText = userData.activeLevel.toString(); } document.getElementById('u-refs').innerText = userData.personalReferralsCount.toString(); const bal = await usdtContract.balanceOf(addr); document.getElementById('u-usdt').innerText = ethers.formatUnits(bal, 6); } catch (e) { console.error("Ошибка загрузки данных профиля:", e); } } // Принудительный вызов при клике document.getElementById('connectBtn').addEventListener('click', (e) => { e.preventDefault(); connect(); });
